999. Available Captures for Rook
Easy
On an 8 x 8 chessboard, there is one white rook.  There also may be empty squares, white bishops, and black pawns.  These are given as characters 'R', '.', 'B', and 'p' respectively. Uppercase characters represent white pieces, and lowercase characters represent black pieces.
The rook moves as in the rules of Chess: it chooses one of four cardinal directions (north, east, west, and south), then moves in that direction until it chooses to stop, reaches the edge of the board, or captures an opposite colored pawn by moving to the same square it occupies.  Also, rooks cannot move into the same square as other friendly bishops.
Return the number of pawns the rook can capture in one move.
 
Example 1:
Input: [[".",".",".",".",".",".",".","."],[".",".",".","p",".",".",".","."],[".",".",".","R",".",".",".","p"],[".",".",".",".",".",".",".","."],[".",".",".",".",".",".",".","."],[".",".",".","p",".",".",".","."],[".",".",".",".",".",".",".","."],[".",".",".",".",".",".",".","."]]Output: 3Explanation:
In this example the rook is able to capture all the pawns.
Example 2:
Input: [[".",".",".",".",".",".",".","."],[".","p","p","p","p","p",".","."],[".","p","p","B","p","p",".","."],[".","p","B","R","B","p",".","."],[".","p","p","B","p","p",".","."],[".","p","p","p","p","p",".","."],[".",".",".",".",".",".",".","."],[".",".",".",".",".",".",".","."]]Output: 0Explanation:
Bishops are blocking the rook to capture any pawn.
Example 3:
Input: [[".",".",".",".",".",".",".","."],[".",".",".","p",".",".",".","."],[".",".",".","p",".",".",".","."],["p","p",".","R",".","p","B","."],[".",".",".",".",".",".",".","."],[".",".",".","B",".",".",".","."],[".",".",".","p",".",".",".","."],[".",".",".",".",".",".",".","."]]Output: 3Explanation:
The rook can capture the pawns at positions b5, d6 and f5.
 
Note:
  1. board.length == board[i].length == 8
  2. board[i][j] is either 'R', '.', 'B', or 'p'
  3. There is exactly one cell with board[i][j] == 'R'

class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
    int z = 0, y = 0, u = 0, d = 0;
    int col = 0, row = 0;
    for (int i = 0; i<8; ++i)
    {
        for (int j = 0; j<8; ++j) {
            if (board[i][j] == *"R")
            {
                col=j,row=i;
            }
        }
    }
    //left
    for(int i=0;i<col;i++)
    {
        if(board[row][i]==*"p") {
            z=1;
        }
        else if(board[row][i]==*"B")
        {
            z=0;
        }
    }
        
    //right
    for(int i=col+1;i<8;i++)
    {
        if(board[row][i]==*"p")
        {
            y=1;break;
        }
        else if(board[row][i]==*"B")
        {
            y=0;break;
        }
    }
        
    //up
    for(int i=0;i<row;i++)
    {
        if(board[i][col]==*"p") {u=1;}
        else if(board[i][col]==*"B")
        {
            u=0;
        }
    }
        
    //down
    for(int i=row+1;i<8;i++)
    {
        if(board[i][col]==*"p")
        {
            d=1;break;
        }
        else if(board[i][col]==*"B") {
            d=0;break;
        }
    }
    int ans = z + y + u + d;
    return ans;
    }
};
