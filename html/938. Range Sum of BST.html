938. Range Sum of BST
Medium
Given the root node of a binary search tree, return the sum of values of all nodes with value between L and R (inclusive).
The binary search tree is guaranteed to have unique values.
 
Example 1:
Input: root = [10,5,15,3,7,null,18], L = 7, R = 15Output: 32
Example 2:
Input: root = [10,5,15,3,7,13,18,1,null,6], L = 6, R = 10Output: 23
 
Note:
  1. The number of nodes in the tree is at most 10000.
  2. The final answer is guaranteed to be less than 2^31.

class Solution {
public:
    
    void in_order(TreeNode* root,vector<int>& list)
    {
        if(root==NULL) return;
        else if(root->left==NULL&&root->right==NULL){
            list.push_back(root->val);
            return;
        }
        in_order(root->left,list);
        list.push_back(root->val);
        in_order(root->right,list);
        
    }
    int rangeSumBST(TreeNode* root, int L, int R) {
        int sum=0;
        vector<int> list;
        in_order(root,list);
        auto first=find(list.begin(),list.end(),L);
        auto Last=find(list.begin(),list.end(),R);
        
        for(auto it=first;it<=Last;++it)
        {
            sum+=*it;
        }
        return sum;
    }
};

class Solution {
public:
    
    int rangeSumBST(TreeNode* root, int L, int R) {
  if (root == nullptr) return 0;
  return (root->val >= L && root->val <= R ? root->val : 0) +
    rangeSumBST(root->left, L, R) + rangeSumBST(root->right, L, R);
    }
};
