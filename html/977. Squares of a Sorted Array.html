977. Squares of a Sorted Array
Easy
Given an array of integers A sorted in non-decreasing order, return an array of the squares of each number, also in sorted non-decreasing order.
 
Example 1:
Input: [-4,-1,0,3,10]Output: [0,1,9,16,100]
Example 2:
Input: [-7,-3,2,3,11]Output: [4,9,9,49,121]
 
Note:
  1. 1 <= A.length <= 10000
  2. -10000 <= A[i] <= 10000
  3. A is sorted in non-decreasing order.
代码1：two pointers
class Solution {
public:
    vector<int> sortedSquares(vector<int>& A) {
    for (auto it = A.begin(); it != A.end(); ++it) {
        *it *= *it;
    }
    vector<int> B(A.size(), 0);

    int left = 0;
    int right = A.size() - 1;

    for (int i = B.size() - 1; i >= 0; --i) {
        if (A[left] > A[right]) {
            B[i] = A[left++];
        }
        else {
            B[i] = A[right--];
        }
    }
    return B;
    }
};

代码2: 
We find the first non-negative element in the array. From there, we move 'negative' pointer (np) towards zero, and 'positive' pointer (pp) towards the end of the array, selecting the next square.
//inside-out
vector<int> sortedSquares(vector<int>& A) {
  vector<int> res;
  int pvt = lower_bound(begin(A), end(A), 0) - begin(A);
  for (int pp = pvt, pn = pvt - 1; pp < A.size() || pn >= 0;) {
    if (abs(pn >= 0 ? A[pn] : INT_MAX) < abs(pp < A.size() ? A[pp] : INT_MAX))
      res.push_back(pow(A[pn--], 2));
    else res.push_back(pow(A[pp++], 2));
  }
  return res;
}

//outside-in
vector<int> sortedSquares(vector<int>& A) {
  vector<int> res(A.size());
  for (int pn = 0, pp = A.size() - 1, pos = A.size() - 1; pn <= pp; --pos)
    res[pos] = pow(abs(A[pn]) < abs(A[pp]) ? A[pp--] : A[pn++], 2);
  return res;
}
